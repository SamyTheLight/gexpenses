document.addEventListener("DOMContentLoaded", function() {
  var totalRepartir = Number(document.getElementById("totalRepartir").dataset.total);
  var totalFaltante = totalRepartir;

  document.getElementById("botonRepartir").addEventListener("click", function(){
      var cantidad = totalRepartir / document.getElementsByClassName('inputCantidad').length;
      Array.from(document.getElementsByClassName('inputCantidad')).forEach(function(input) {
          input.value = cantidad;
      });
      totalFaltante = 0;
      document.getElementById('totalFaltante').innerText = 'Cantidad faltante: ' + totalFaltante;
      document.getElementById('btnEnviar').style.display = 'block'; // Mostrar el botón de envío cuando se haya repartido todo
  });

  Array.from(document.getElementsByClassName("inputCantidad")).forEach(function(input) {
      input.addEventListener("change", function(){
          totalFaltante = totalRepartir;
          Array.from(document.getElementsByClassName("inputCantidad")).forEach(function(input) {
              totalFaltante -= Number(input.value);
          });
          document.getElementById('totalFaltante').innerText = 'Cantidad faltante: ' + totalFaltante;
          if (totalFaltante === 0) {
              document.getElementById("btnEnviar").style.display = 'block'; // Mostrar el botón de envío cuando se haya repartido todo
          } else {
              document.getElementById("btnEnviar").style.display = 'none'; // Ocultar el botón de envío si aún queda algo por repartir
          }
      });
  });

  document.getElementById("act-form1").addEventListener("submit", function(event) {
      if (totalFaltante !== 0) {
          alert('Las cantidades no cuadran. Por favor revise.');
          event.preventDefault();
          return false;
      }
      return true;
  });
});
